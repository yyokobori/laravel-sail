# メインクラス実装サンプル

参照:
- 命名規約: `ai/03_react.md`
- 実装規約: `ai/03.02_class.md`

```tsx
import React from 'react';
import { useNavigate } from 'react-router-dom';

import { PublicTemplate } from '../template/PublicTemplate';
import './sample.css';

type SamplePageProps = {
  id: string;
  mode: 'view' | 'edit';
  status: 'active' | 'archived';
  className?: string;
  onSubmit?: (id: string) => void;
};

const PAGE_TITLE = 'Sample Page';

/**
 * 画面タイトルを生成する。
 * @param id 識別子
 * @returns 画面タイトル
 */
function makePageTitle(id: string): string {
  return `${PAGE_TITLE} - ${id}`;
}

/**
 * サンプル画面を描画する。
 * @param props 画面表示設定
 * @returns サンプル画面
 */
function SamplePage(props: SamplePageProps): JSX.Element {
  const navigate = useNavigate();

  const handleSubmit = (): void => {
    props.onSubmit?.(props.id);
    navigate('/');
  };

  return (
    <PublicTemplate>
      <main className={props.className}>
        <h1>{makePageTitle(props.id)}</h1>
        <p>{props.mode}</p>
        <p>{props.status}</p>
        <button onClick={handleSubmit}>submit</button>
      </main>
    </PublicTemplate>
  );
}

export { SamplePage };
```

## サンプル要点
- 本サンプルは `ai/03.02_class.md` の規約を反映した最小例である。
